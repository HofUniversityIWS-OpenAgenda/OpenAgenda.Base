 123      * Loads the metadata for the specified class into the provided container.
 124      *
 125      * @param string $className
 126      * @param \Doctrine\Common\Persistence\Mapping\ClassMetadata $metadata
 127      * @return void
 128      * @throws \Doctrine\ORM\Mapping\MappingException
 129      * @throws \UnexpectedValueException
 130      * @todo adjust when Doctrine 2 supports value objects, see http://www.doctrine-project.org/jira/browse/DDC-93
 131      */
 132     public function loadMetadataForClass($className, \Doctrine\Common\Persistence\Mapping\ClassMetadata $metadata) {
 133         $class = $metadata->getReflectionClass();
 134         $classSchema = $this->getClassSchema($class->getName());
 135         $classAnnotations = $this->reader->getClassAnnotations($class);
 136 
 137         // Evaluate Entity annotation
 138         if (isset($classAnnotations['Doctrine\ORM\Mapping\MappedSuperclass'])) {
 139             $mappedSuperclassAnnotation = $classAnnotations['Doctrine\ORM\Mapping\MappedSuperclass'];
 140             if ($mappedSuperclassAnnotation->repositoryClass !== NULL) {
 141                 $metadata->setCustomRepositoryClass($mappedSuperclassAnnotation->repositoryClass);
 142             }
 143             $metadata->isMappedSuperclass = TRUE;
 144         } elseif (isset($classAnnotations['TYPO3\Flow\Annotations\Entity']) || isset($classAnnotations['Doctrine\ORM\Mapping\Entity'])) {
 145             $entityAnnotation = isset($classAnnotations['TYPO3\Flow\Annotations\Entity']) ? $classAnnotations['TYPO3\Flow\Annotations\Entity'] : $classAnnotations['Doctrine\ORM\Mapping\Entity'];
 146             if ($entityAnnotation->repositoryClass !== NULL) {
 147                 $metadata->setCustomRepositoryClass($entityAnnotation->repositoryClass);
 148             } elseif ($classSchema->getRepositoryClassName() !== NULL) {
 149                 if ($this->reflectionService->isClassImplementationOf($classSchema->getRepositoryClassName(), 'Doctrine\ORM\EntityRepository')) {
 150                     $metadata->setCustomRepositoryClass($classSchema->getRepositoryClassName());
 151                 }
 152             }
 153             if ($entityAnnotation->readOnly) {
 154                 $metadata->markReadOnly();
 155             }
 156         } elseif ($classSchema->getModelType() === \TYPO3\Flow\Reflection\ClassSchema::MODELTYPE_VALUEOBJECT) {
 157             // also ok... but we make it read-only
 158             $metadata->markReadOnly();
 159         } else {
 160             throw \Doctrine\ORM\Mapping\MappingException::classIsNotAValidEntityOrMappedSuperClass($className);
 161         }
 162 
 163         // Evaluate Table annotation
 164         $primaryTable = array();
 165         if (isset($classAnnotations['Doctrine\ORM\Mapping\Table'])) {
 166             $tableAnnotation = $classAnnotations['Doctrine\ORM\Mapping\Table'];
 167             $primaryTable['name'] = $tableAnnotation->name;
 168             $primaryTable['schema'] = $tableAnnotation->schema;
 169 
 170             if ($tableAnnotation->indexes !== NULL) {
 171                 foreach ($tableAnnotation->indexes as $indexAnnotation) {
 172                     $index = array('columns' => $indexAnnotation->columns);
 173                     if (!empty($indexAnnotation->name)) {
 174                         $primaryTable['indexes'][$indexAnnotation->name] = $index;
 175                     } else {
 176                         $primaryTable['indexes'][] = $index;
 177                     }
 178                 }
 179             }
 180 
 181             if ($tableAnnotation->uniqueConstraints !== NULL) {
 182                 foreach ($tableAnnotation->uniqueConstraints as $uniqueConstraint) {
 183                     $uniqueConstraint = array('columns' => $uniqueConstraint->columns);
 184                     if (!empty($uniqueConstraint->name)) {
 185                         $primaryTable['uniqueConstraints'][$uniqueConstraint->name] = $uniqueConstraint;
 186                     } else {
 187                         $primaryTable['uniqueConstraints'][] = $uniqueConstraint;
 188                     }
 189                 }
 190             }
 191 
 192             if ($tableAnnotation->options !== NULL) {
 193                 $primaryTable['options'] = $tableAnnotation->options;
 194             }
 195         }
 196         if (!isset($primaryTable['name'])) {
 197             $className = $classSchema->getClassName();
 198             $primaryTable['name'] = $this->inferTableNameFromClassName($className);
 199         }
 200 
 201         // Evaluate NamedNativeQueries annotation
 202         if (isset($classAnnotations['Doctrine\ORM\Mapping\NamedNativeQueries'])) {
 203             $namedNativeQueriesAnnotation = $classAnnotations['Doctrine\ORM\Mapping\NamedNativeQueries'];
 204 
 205             foreach ($namedNativeQueriesAnnotation->value as $namedNativeQuery) {
 206                 $metadata->addNamedNativeQuery(array(
 207                     'name' => $namedNativeQuery->name,
 208                     'query' => $namedNativeQuery->query,
 209                     'resultClass' => $namedNativeQuery->resultClass,

 234                     }

 274             $inheritanceTypeAnnotation = $classAnnotations['Doctrine\ORM\Mapping\InheritanceType'];
 275             $inheritanceType = constant('Doctrine\ORM\Mapping\ClassMetadata::INHERITANCE_TYPE_' . strtoupper($inheritanceTypeAnnotation->value));
 276 
 277             if ($inheritanceType !== \Doctrine\ORM\Mapping\ClassMetadata::INHERITANCE_TYPE_NONE) {
 278 
 279                 // Evaluate DiscriminatorColumn annotation
 280                 if (isset($classAnnotations['Doctrine\ORM\Mapping\DiscriminatorColumn'])) {
 281                     $discriminatorColumnAnnotation = $classAnnotations['Doctrine\ORM\Mapping\DiscriminatorColumn'];
 282                     $discriminatorColumn = array(
 283                         'name' => $discriminatorColumnAnnotation->name,
 284                         'type' => $discriminatorColumnAnnotation->type,
 285                         'length' => $discriminatorColumnAnnotation->length,
 286                         'columnDefinition' => $discriminatorColumnAnnotation->columnDefinition

 292                 }

 296                     $discriminatorMapAnnotation = $classAnnotations['Doctrine\ORM\Mapping\DiscriminatorMap'];
 297                     $discriminatorMap = $discriminatorMapAnnotation->value;
 298                 } else {

 367     /**
 368      * Truncate an identifier if needed and append a hash to ensure uniqueness.
 369      *
 370      * @param string $identifier
 371      * @param integer $lengthLimit
 372      * @param string $hashSource
 373      * @return string
 374      */
 375     protected function truncateIdentifier($identifier, $lengthLimit = NULL, $hashSource = NULL) {
 376         if ($lengthLimit === NULL) {
 377             $lengthLimit = $this->getMaxIdentifierLength();
 378         }
 379         if (strlen($identifier) > $lengthLimit) {
 380             $identifier = substr($identifier, 0, $lengthLimit - 6) . '_' . substr(sha1($hashSource !== NULL ? $hashSource : $identifier), 0, 5);
 381         }
 382         return $identifier;
 383     }

 385     /**
 386      * Given a class and property name a table name is returned. That name should be reasonably unique.
 387      *
 388      * @param string $className Model class name the table corresponds to
 389      * @param string $propertyName Name of the property to be joined
 390      * @return string Truncated database table name
 391      */
 392     protected function inferJoinTableNameFromClassAndPropertyName($className, $propertyName) {
 393         $prefix = $this->inferTableNameFromClassName($className);
 394         $suffix = '_' . strtolower($propertyName . '_join');
 395 
 396         // In order to keep backwards compatibility with earlier versions, truncate the table name in two steps:
 397         if (strlen($prefix . $suffix) > $this->getMaxIdentifierLength()) {
 398             $prefix = $this->inferTableNameFromClassName($className, $this->getMaxIdentifierLength() - strlen($suffix));
 399         }
 400         // Truncate a second time if the property name was too long as well:
 401         if (strlen($prefix . $suffix) > $this->getMaxIdentifierLength()) {
 402             return $this->truncateIdentifier($prefix . $suffix, $this->getMaxIdentifierLength());
 403         } else {
 404             return $prefix . $suffix;
 405         }
 406     }

 408     /**
 409      * Build a name for a column in a jointable.
 410      *
 411      * @param string $className
 412      * @return string
 413      */
 414     protected function buildJoinTableColumnName($className) {
 415         if (preg_match('/^(?P<PackageNamespace>\w+(?:\\\\\w+)*)\\\\Domain\\\\Model\\\\(?P<ModelNamePrefix>(\w+\\\\)?)(?P<ModelName>\w+)$/', $className, $matches)) {
 416             $packageNamespaceParts = explode('\\', $matches['PackageNamespace']);
 417             $tableName = strtolower(strtr($packageNamespaceParts[count($packageNamespaceParts) - 1], '\\', '_') . ($matches['ModelNamePrefix'] !== '' ? '_' . strtr(rtrim($matches['ModelNamePrefix'], '\\'), '\\', '_') : '') . '_' . $matches['ModelName']);
 418         } else {
 419             $classNameParts = explode('\\', $className);
 420             $tableName = strtolower($classNameParts[1] . '_' . implode('_', array_slice($classNameParts, -2, 2)));
 421         }
 422 
 423         return $this->truncateIdentifier($tableName);
 424     }

 433      * @param integer $direction regular or inverse mapping (use is to be coded)
 434      * @return array
 435      */
 436     protected function buildJoinColumnsIfNeeded(array $joinColumns, array $mapping, \ReflectionProperty $property, $direction = self::MAPPING_REGULAR) {
 437         if ($joinColumns === array()) {
 438             $joinColumns[] = array(
 439                 'name' => strtolower($property->getName()),
 440                 'referencedColumnName' => NULL,
 441             );
 442         }
 443         foreach ($joinColumns as &$joinColumn) {
 444             if ($joinColumn['referencedColumnName'] === NULL || $joinColumn['referencedColumnName'] === 'id') {
 445                 if ($direction === self::MAPPING_REGULAR) {
 446                     $idProperties = $this->reflectionService->getPropertyNamesByTag($mapping['targetEntity'], 'id');
 447                     $joinColumnName = $this->buildJoinTableColumnName($mapping['targetEntity']);
 448                 } else {
 449                     $className = preg_replace('/' . \TYPO3\Flow\Object\Proxy\Compiler::ORIGINAL_CLASSNAME_SUFFIX . '$/', '', $property->getDeclaringClass()->getName());
 450                     $idProperties = $this->reflectionService->getPropertyNamesByTag($className, 'id');
 451                     $joinColumnName = $this->buildJoinTableColumnName($className);
 452                 }
 453                 if (count($idProperties) === 0) {
 454                     $joinColumn['name'] = $joinColumn['name'] === NULL ? $joinColumnName : $joinColumn['name'];
 455                     $joinColumn['referencedColumnName'] = strtolower('Persistence_Object_Identifier');
 456                 } elseif (count($idProperties) === 1) {
 457                     $joinColumn['name'] = $joinColumn['name'] === NULL ? $joinColumnName : $joinColumn['name'];
 458                     $joinColumn['referencedColumnName'] = strtolower(current($idProperties));
 459                 }
 460             }
 461         }
 462 
 463         return $joinColumns;
 464     }

 466     /**
 467      * Evaluate the property annotations and amend the metadata accordingly.
 468      *
 469      * @param \Doctrine\ORM\Mapping\ClassMetadataInfo $metadata
 470      * @return void
 471      * @throws \Doctrine\ORM\Mapping\MappingException
 472      */
 473     protected function evaluatePropertyAnnotations(\Doctrine\ORM\Mapping\ClassMetadataInfo $metadata) {
 474         $className = $metadata->name;
 475 
 476         $class = $metadata->getReflectionClass();
 477         $classSchema = $this->getClassSchema($className);
 478 
 479         foreach ($class->getProperties() as $property) {
 480             if (!$classSchema->hasProperty($property->getName())
 481                     || $classSchema->isPropertyTransient($property->getName())
 482                     || $metadata->isMappedSuperclass && !$property->isPrivate()
 483                     || $metadata->isInheritedField($property->getName())
 484                     || $metadata->isInheritedAssociation($property->getName())) {
 485                 continue;
 486             }
 487 
 488             $propertyMetaData = $classSchema->getProperty($property->getName());
 489 
 490             $mapping = array();
 491             $mapping['fieldName'] = $property->getName();
 492             $mapping['columnName'] = strtolower($property->getName());
 493             $mapping['targetEntity'] = $propertyMetaData['type'];
 494 
 495             $joinColumns = $this->evaluateJoinColumnAnnotations($property);
 496 
 497             // Field can only be annotated with one of:
 498             // @OneToOne, @OneToMany, @ManyToOne, @ManyToMany, @Column (optional)
 499             if ($oneToOneAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\OneToOne')) {
 500                 if ($oneToOneAnnotation->targetEntity) {
 501                     $mapping['targetEntity'] = $oneToOneAnnotation->targetEntity;
 502                 }
 503                 $mapping['joinColumns'] = $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property);
 504                 $mapping['mappedBy'] = $oneToOneAnnotation->mappedBy;
 505                 $mapping['inversedBy'] = $oneToOneAnnotation->inversedBy;
 506                 if ($oneToOneAnnotation->cascade) {
 507                     $mapping['cascade'] = $oneToOneAnnotation->cascade;
 508                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 509                     $mapping['cascade'] = array('all');
 510                 }
 511                 if ($oneToOneAnnotation->orphanRemoval) {
 512                     $mapping['orphanRemoval'] = $oneToOneAnnotation->orphanRemoval;
 513                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 514                     $mapping['orphanRemoval'] = TRUE;
 515                 }
 516                 $mapping['fetch'] = $this->getFetchMode($className, $oneToOneAnnotation->fetch);
 517                 $metadata->mapOneToOne($mapping);
 518             } elseif ($oneToManyAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\OneToMany')) {
 519                 $mapping['mappedBy'] = $oneToManyAnnotation->mappedBy;
 520                 if ($oneToManyAnnotation->targetEntity) {
 521                     $mapping['targetEntity'] = $oneToManyAnnotation->targetEntity;
 522                 } elseif (isset($propertyMetaData['elementType'])) {
 523                     $mapping['targetEntity'] = $propertyMetaData['elementType'];
 524                 }
 525                 if ($oneToManyAnnotation->cascade) {
 526                     $mapping['cascade'] = $oneToManyAnnotation->cascade;
 527                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 528                     $mapping['cascade'] = array('all');
 529                 }
 530                 if ($oneToManyAnnotation->orphanRemoval) {
 531                     $mapping['orphanRemoval'] = $oneToManyAnnotation->orphanRemoval;
 532                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 533                     $mapping['orphanRemoval'] = TRUE;
 534                 }
 535                 $mapping['fetch'] = $this->getFetchMode($className, $oneToManyAnnotation->fetch);
 536 
 537                 if ($orderByAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\OrderBy')) {
 538                     $mapping['orderBy'] = $orderByAnnotation->value;
 539                 }
 540 
 541                 $metadata->mapOneToMany($mapping);
 542             } elseif ($manyToOneAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\ManyToOne')) {
 543                 if ($manyToOneAnnotation->targetEntity) {
 544                     $mapping['targetEntity'] = $manyToOneAnnotation->targetEntity;
 545                 }
 546                 $mapping['joinColumns'] = $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property);
 547                 if ($manyToOneAnnotation->cascade) {
 548                     $mapping['cascade'] = $manyToOneAnnotation->cascade;
 549                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 550                     $mapping['cascade'] = array('all');
 551                 }
 552                 $mapping['inversedBy'] = $manyToOneAnnotation->inversedBy;
 553                 $mapping['fetch'] = $this->getFetchMode($className, $manyToOneAnnotation->fetch);
 554                 $metadata->mapManyToOne($mapping);
 555             } elseif ($manyToManyAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\ManyToMany')) {
 556                 if ($manyToManyAnnotation->targetEntity) {
 557                     $mapping['targetEntity'] = $manyToManyAnnotation->targetEntity;
 558                 } elseif (isset($propertyMetaData['elementType'])) {
 559                     $mapping['targetEntity'] = $propertyMetaData['elementType'];
 560                 }
 561                 if ($joinTableAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\JoinTable')) {
 562                     $joinTable = $this->evaluateJoinTableAnnotation($joinTableAnnotation, $property, $className, $mapping);
 563                 } else {
 564                     $joinColumns = array(
 565                         array(
 566                             'name' => NULL,
 567                             'referencedColumnName' => NULL,
 568                         )
 569                     );
 570 
 571                     $joinTable = array(
 572                         'name' => $this->inferJoinTableNameFromClassAndPropertyName($className, $property->getName()),
 573                         'joinColumns' => $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property, self::MAPPING_MM_REGULAR),
 574                         'inverseJoinColumns' => $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property)
 575                     );
 576                 }
 577 
 578                 $mapping['joinTable'] = $joinTable;
 579                 $mapping['mappedBy'] = $manyToManyAnnotation->mappedBy;
 580                 $mapping['inversedBy'] = $manyToManyAnnotation->inversedBy;
 581                 if ($manyToManyAnnotation->cascade) {
 582                     $mapping['cascade'] = $manyToManyAnnotation->cascade;
 583                 } elseif ($this->isAggregateRoot($mapping['targetEntity'], $className) === FALSE) {
 584                     $mapping['cascade'] = array('all');
 585                 }
 586 
 587                 $mapping['orphanRemoval'] = $manyToManyAnnotation->orphanRemoval;
 588                 $mapping['fetch'] = $this->getFetchMode($className, $manyToManyAnnotation->fetch);
 589 
 590                 if ($orderByAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\OrderBy')) {
 591                     $mapping['orderBy'] = $orderByAnnotation->value;
 592                 }
 593 
 594                 $metadata->mapManyToMany($mapping);
 595             } else {
 596                 $mapping['nullable'] = FALSE;
 597 
 598                 if ($columnAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\Column')) {
 599                     $mapping = $this->addColumnToMappingArray($columnAnnotation, $mapping);
 600                 }
 601 
 602                 if (!isset($mapping['type'])) {
 603                     switch ($propertyMetaData['type']) {
 604                         case 'DateTime':
 605                             $mapping['type'] = 'datetime';
 606                             break;
 607                         case 'string':
 608                         case 'integer':
 609                         case 'boolean':
 610                         case 'float':
 611                         case 'array':
 612                             $mapping['type'] = $propertyMetaData['type'];
 613                             break;
 614                         default:
 615                             if (strpos($propertyMetaData['type'], '\\') !== FALSE) {
 616                                 if ($this->reflectionService->isClassAnnotatedWith($propertyMetaData['type'], 'TYPO3\Flow\Annotations\ValueObject')) {
 617                                     $mapping['type'] = 'object';
 618                                 } elseif (class_exists($propertyMetaData['type'])) {
 619 
 620                                     throw \Doctrine\ORM\Mapping\MappingException::missingRequiredOption($property->getName(), 'OneToOne', sprintf('The property "%s" in class "%s" has a non standard data type and doesn\'t define the type of the relation. You have to use one of these annotations: @OneToOne, @OneToMany, @ManyToOne, @ManyToMany', $property->getName(), $className));
 621                                 }
 622                             } else {
 623                                 throw \Doctrine\ORM\Mapping\MappingException::propertyTypeIsRequired($className, $property->getName());
 624                             }
 625                     }
 626 
 627                 }
 628 
 629                 if ($this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\Id') !== NULL) {
 630                     $mapping['id'] = TRUE;
 631                 }
 632 
 633                 if ($generatedValueAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\GeneratedValue')) {
 634                     $metadata->setIdGeneratorType(constant('Doctrine\ORM\Mapping\ClassMetadata::GENERATOR_TYPE_' . strtoupper($generatedValueAnnotation->strategy)));
 635                 }
 636 
 637                 if ($this->reflectionService->isPropertyAnnotatedWith($className, $property->getName(), 'Doctrine\ORM\Mapping\Version')) {
 638                     $metadata->setVersionMapping($mapping);
 639                 }
 640 
 641                 $metadata->mapField($mapping);
 642 
 643                 // Check for SequenceGenerator/TableGenerator definition
 644                 if ($seqGeneratorAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\SequenceGenerator')) {
 645                     $metadata->setSequenceGeneratorDefinition(array(
 646                         'sequenceName' => $seqGeneratorAnnotation->sequenceName,
 647                         'allocationSize' => $seqGeneratorAnnotation->allocationSize,
 648                         'initialValue' => $seqGeneratorAnnotation->initialValue
 649                     ));
 650                 } elseif ($this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\TableGenerator') !== NULL) {
 651                     throw \Doctrine\ORM\Mapping\MappingException::tableIdGeneratorNotImplemented($className);
 652                 } elseif ($customGeneratorAnnotation = $this->reader->getPropertyAnnotation($property, 'Doctrine\ORM\Mapping\CustomIdGenerator')) {
 653                     $metadata->setCustomGeneratorDefinition(array(
 654                         'class' => $customGeneratorAnnotation->class
 655                     ));
 656                 }
 657             }
 658 
 659         }
 660     }

 662     /**
 663      * Evaluate JoinTable annotations and fill missing bits as needed.
 664      *
 665      * @param \Doctrine\ORM\Mapping\JoinTable $joinTableAnnotation
 666      * @param \ReflectionProperty $property
 667      * @param string $className
 668      * @param array $mapping
 669      * @return array
 670      */
 671     protected function evaluateJoinTableAnnotation(\Doctrine\ORM\Mapping\JoinTable $joinTableAnnotation, \ReflectionProperty $property, $className, array $mapping) {
 672         $joinTable = array(
 673             'name' => $joinTableAnnotation->name,
 674             'schema' => $joinTableAnnotation->schema
 675         );
 676         if ($joinTable['name'] === NULL) {
 677             $joinTable['name'] = $this->inferJoinTableNameFromClassAndPropertyName($className, $property->getName());
 678         }
 679 
 680         foreach ($joinTableAnnotation->joinColumns as $joinColumn) {
 681             $joinTable['joinColumns'][] = $this->joinColumnToArray($joinColumn);
 682         }
 683         if (array_key_exists('joinColumns', $joinTable)) {
 684             $joinTable['joinColumns'] = $this->buildJoinColumnsIfNeeded($joinTable['joinColumns'], $mapping, $property, self::MAPPING_MM_REGULAR);
 685         } else {
 686             $joinColumns = array(
 687                 array(
 688                     'name' => NULL,
 689                     'referencedColumnName' => NULL,
 690                 )
 691             );
 692             $joinTable['joinColumns'] = $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property, self::MAPPING_MM_REGULAR);
 693         }
 694 
 695         foreach ($joinTableAnnotation->inverseJoinColumns as $joinColumn) {
 696             $joinTable['inverseJoinColumns'][] = $this->joinColumnToArray($joinColumn);
 697         }
 698         if (array_key_exists('inverseJoinColumns', $joinTable)) {
 699             $joinTable['inverseJoinColumns'] = $this->buildJoinColumnsIfNeeded($joinTable['inverseJoinColumns'], $mapping, $property);
 700         } else {
 701             $joinColumns = array(
 702                 array(
 703                     'name' => NULL,
 704                     'referencedColumnName' => NULL,
 705                 )
 706             );
 707             $joinTable['inverseJoinColumns'] = $this->buildJoinColumnsIfNeeded($joinColumns, $mapping, $property);
 708         }
 709 
 710         return $joinTable;
 711     }

 735     /**
 736      * Evaluate the association overrides annotations and amend the metadata accordingly.
 737      *
 738      * @param array $classAnnotations
 739      * @param \Doctrine\ORM\Mapping\ClassMetadataInfo $metadata
 740      * @return void
 741      */
 742     protected function evaluateOverridesAnnotations(array $classAnnotations, \Doctrine\ORM\Mapping\ClassMetadataInfo $metadata) {
 743         if (isset($classAnnotations['Doctrine\ORM\Mapping\AssociationOverrides'])) {
 744             $associationOverridesAnnotation = $classAnnotations['Doctrine\ORM\Mapping\AssociationOverrides'];
 745 
 746             foreach ($associationOverridesAnnotation->value as $associationOverride) {
 747                 $override = array();
 748                 $fieldName = $associationOverride->name;
 749 
 750                 // Check for JoinColummn/JoinColumns annotations
 751                 if ($associationOverride->joinColumns) {
 752                     $joinColumns = array();
 753                     foreach ($associationOverride->joinColumns as $joinColumn) {
 754                         $joinColumns[] = $this->joinColumnToArray($joinColumn);
 755                     }
 756                     $override['joinColumns'] = $joinColumns;
 757                 }
 758 
 759                 // Check for JoinTable annotations
 760                 if ($associationOverride->joinTable) {
 761                     $joinTable = NULL;
 762                     $joinTableAnnotation = $associationOverride->joinTable;
 763                     $joinTable = array(
 764                         'name' => $joinTableAnnotation->name,
 765                         'schema' => $joinTableAnnotation->schema
 766                     );
 767 
 768                     foreach ($joinTableAnnotation->joinColumns as $joinColumn) {
 769                         $joinTable['joinColumns'][] = $this->joinColumnToArray($joinColumn);
 770                     }
 771 
 772                     foreach ($joinTableAnnotation->inverseJoinColumns as $joinColumn) {
 773                         $joinTable['inverseJoinColumns'][] = $this->joinColumnToArray($joinColumn);
 774                     }
 775 
 776                     $override['joinTable'] = $joinTable;
 777                 }
 778 
 779                 $metadata->setAssociationOverride($fieldName, $override);
 780             }
 781         }
 782 
 783         // Evaluate AttributeOverrides annotation
 784         if (isset($classAnnotations['Doctrine\ORM\Mapping\AttributeOverrides'])) {
 785             $attributeOverridesAnnotation = $classAnnotations['Doctrine\ORM\Mapping\AttributeOverrides'];
 786             foreach ($attributeOverridesAnnotation->value as $attributeOverrideAnnotation) {
 787                 $attributeOverride = $this->addColumnToMappingArray($attributeOverrideAnnotation->column, array(), $attributeOverrideAnnotation->name);
 788                 $metadata->setAttributeOverride($attributeOverrideAnnotation->name, $attributeOverride);
 789             }
 790         }
 791     }

 971     /**
 972      * Checks if the specified class has a property annotated with Id
 973      *
 974      * @param string $className Name of the class to check against
 975      * @param string $methodName Name of the method to check against
 976      * @param string $methodDeclaringClassName Name of the class the method was originally declared in
 977      * @param mixed $pointcutQueryIdentifier Some identifier for this query - must at least differ from a previous identifier. Used for circular reference detection.
 978      * @return boolean TRUE if the class has *no* Id properties
 979      */
 980     public function matches($className, $methodName, $methodDeclaringClassName, $pointcutQueryIdentifier) {

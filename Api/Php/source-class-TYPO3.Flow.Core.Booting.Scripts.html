 91     static public function forceFlushCachesIfNecessary(Bootstrap $bootstrap) {
 92         if (!isset($_SERVER['argv']) || !isset($_SERVER['argv'][1]) || !isset($_SERVER['argv'][2])
 93             || !in_array($_SERVER['argv'][1], array('typo3.flow:cache:flush', 'flow:cache:flush'))

101 
102         echo 'Force-flushed caches for "' . $bootstrap->getContext() . '" context.' . PHP_EOL;
103         exit(0);
104     }

235     /**
236      * Runs the compile step if necessary
237      *
238      * @param Bootstrap $bootstrap
239      * @return void
240      * @throws \TYPO3\Flow\Exception
241      */
242     static public function initializeProxyClasses(Bootstrap $bootstrap) {
243         $objectConfigurationCache = $bootstrap->getEarlyInstance('TYPO3\Flow\Cache\CacheManager')->getCache('Flow_Object_Configuration');
244 
245         $configurationManager = $bootstrap->getEarlyInstance('TYPO3\Flow\Configuration\ConfigurationManager');
246         $settings = $configurationManager->getConfiguration(\TYPO3\Flow\Configuration\ConfigurationManager::CONFIGURATION_TYPE_SETTINGS, 'TYPO3.Flow');
247 
248         // The compile sub command will only be run if the code cache is completely empty:
249         if ($objectConfigurationCache->has('allCompiledCodeUpToDate') === FALSE) {
250             self::executeCommand('typo3.flow:core:compile', $settings);
251             if (isset($settings['persistence']['doctrine']['enable']) && $settings['persistence']['doctrine']['enable'] === TRUE) {
252                 self::compileDoctrineProxies($bootstrap);
253             }
254         }
255 
256         // Check if code was updated, if not something went wrong
257         if ($objectConfigurationCache->has('allCompiledCodeUpToDate') === FALSE) {
258             if (DIRECTORY_SEPARATOR === '/') {
259                 $phpBinaryPathAndFilename = '"' . escapeshellcmd(\TYPO3\Flow\Utility\Files::getUnixStylePath($settings['core']['phpBinaryPathAndFilename'])) . '"';
260             } else {
261                 $phpBinaryPathAndFilename = escapeshellarg(\TYPO3\Flow\Utility\Files::getUnixStylePath($settings['core']['phpBinaryPathAndFilename']));
262             }
263             $command = sprintf('%s -c %s -v', $phpBinaryPathAndFilename, escapeshellarg(php_ini_loaded_file()));
264             exec($command, $output, $result);
265             if ($result !== 0) {
266                 if (!file_exists($phpBinaryPathAndFilename)) {
267                     throw new \TYPO3\Flow\Exception(sprintf('It seems like the PHP binary "%s" cannot be executed by Flow. Set the correct path to the PHP executable in Configuration/Settings.yaml, setting TYPO3.Flow.core.phpBinaryPathAndFilename.', $settings['core']['phpBinaryPathAndFilename']), 1315561483);
268                 }
269                 throw new \TYPO3\Flow\Exception(sprintf('It seems like the PHP binary "%s" cannot be executed by Flow. The command executed was "%s" and returned the following: %s', $settings['core']['phpBinaryPathAndFilename'], $command, PHP_EOL . implode(PHP_EOL, $output)), 1354704332);
270             }
271             echo PHP_EOL . 'Flow: The compile run failed. Please check the error output or system log for more information.' . PHP_EOL;
272             exit(1);
273         }
274     }

507     /**
508      * Executes the given command as a sub-request to the Flow CLI system.
509      *
510      * @param string $commandIdentifier E.g. typo3.flow:cache:flush
511      * @param array $settings The TYPO3.Flow settings
512      * @param boolean $outputResults if FALSE the output of this command is only echoed if the execution was not successful
513      * @param array $commandArguments Command arguments
514      * @return boolean TRUE if the command execution was successful (exit code = 0)
515      * @api
516      * @throws \TYPO3\Flow\Core\Booting\Exception\SubProcessException if execution of the sub process failed
517      */
518     static public function executeCommand($commandIdentifier, array $settings, $outputResults = TRUE, array $commandArguments = array()) {
519         $command = self::buildSubprocessCommand($commandIdentifier, $settings, $commandArguments);
520         $output = array();
521         exec($command, $output, $result);
522         if ($result !== 0) {
523             if (count($output) > 0) {
524                 $exceptionMessage = implode(PHP_EOL, $output);
525             } else {
526                 $exceptionMessage = sprintf('Execution of subprocess failed with exit code %d without any further output. (Please check your PHP error log for possible Fatal errors)', $result);
527             }
528             throw new Exception\SubProcessException($exceptionMessage, 1355480641);
529         }
530         if ($outputResults) {
531             echo implode(PHP_EOL, $output);
532         }
533         return $result === 0;
534     }

536     /**
537      * @param string $commandIdentifier E.g. typo3.flow:cache:flush
538      * @param array $settings The TYPO3.Flow settings
539      * @param array $commandArguments Command arguments
540      *
541      * @return string A command line command ready for being exec()uted
542      */
543     protected static function buildSubprocessCommand($commandIdentifier, $settings, array $commandArguments = array()) {
544         $subRequestEnvironmentVariables = array(
545             'FLOW_ROOTPATH' => FLOW_PATH_ROOT,
546             'FLOW_CONTEXT' => $settings['core']['context']
547         );
548         if (isset($settings['core']['subRequestEnvironmentVariables'])) {
549             $subRequestEnvironmentVariables = array_merge($subRequestEnvironmentVariables, $settings['core']['subRequestEnvironmentVariables']);
550         }
551 
552         $command = '';
553         foreach ($subRequestEnvironmentVariables as $argumentKey => $argumentValue) {
554             if (DIRECTORY_SEPARATOR === '/') {
555                 $command .= sprintf('%s=%s ', $argumentKey, escapeshellarg($argumentValue));
556             } else {
557                 $command .= sprintf('SET %s=%s&', $argumentKey, escapeshellarg($argumentValue));
558             }
559         }
560         if (DIRECTORY_SEPARATOR === '/') {
561             $phpBinaryPathAndFilename = '"' . escapeshellcmd(\TYPO3\Flow\Utility\Files::getUnixStylePath($settings['core']['phpBinaryPathAndFilename'])) . '"';
562         } else {
563             $phpBinaryPathAndFilename = escapeshellarg(\TYPO3\Flow\Utility\Files::getUnixStylePath($settings['core']['phpBinaryPathAndFilename']));
564         }
565         $command .= $phpBinaryPathAndFilename;
566         if (!isset($settings['core']['subRequestPhpIniPathAndFilename']) || $settings['core']['subRequestPhpIniPathAndFilename'] !== FALSE) {
567             if (!isset($settings['core']['subRequestPhpIniPathAndFilename'])) {
568                 $useIniFile = php_ini_loaded_file();
569             } else {
570                 $useIniFile = $settings['core']['subRequestPhpIniPathAndFilename'];
571             }
572             $command .= ' -c ' . escapeshellarg($useIniFile);
573         }
574 
575         $escapedArguments = '';
576         if ($commandArguments !== array()) {
577             foreach ($commandArguments as $argument=>$argumentValue) {
578                 $escapedArguments .= ' ' . escapeshellarg('--' . trim($argument));
579                 if (trim($argumentValue) !== '') {
580                     $escapedArguments .= ' ' . escapeshellarg(trim($argumentValue));
581                 }
582             }
583         }
584 
585         $command .= sprintf(' %s %s %s', escapeshellarg(FLOW_PATH_FLOW . 'Scripts/flow.php'), escapeshellarg($commandIdentifier), trim($escapedArguments));
586 
587         return trim($command);
588     }

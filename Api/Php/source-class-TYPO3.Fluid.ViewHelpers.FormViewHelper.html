 21 /**
 22  * Used to output an HTML <form> tag which is targeted at the specified action, in the current controller and package.
 23 
 24  * = Examples =
 25  *
 26  * <code title="Basic usage, POST method">
 27  * <f:form action="...">...</f:form>
 28  * </code>
 29  * <output>
 30  * <form action="...">...</form>
 31  * </output>
 32  *
 33  * <code title="Basic usage, GET method">
 34  * <f:form action="..." method="get">...</f:form>
 35  * </code>
 36  * <output>
 37  * <form method="GET" action="...">...</form>
 38  * </output>
 39  *
 40  * <code title="Form with a sepcified encoding type">
 41  * <f:form action=".." controller="..." package="..." enctype="multipart/form-data">...</f:form>
 42  * </code>
 43  * <output>
 44  * <form enctype="multipart/form-data" action="...">...</form>
 45  * </output>
 46  *
 47  * <code title="Binding a domain object to a form">
 48  * <f:form action="..." name="customer" object="{customer}">
 49  *   <f:form.hidden property="id" />
 50  *   <f:form.textfield property="name" />
 51  * </f:form>
 52  * </code>
 53  * <output>
 54  * A form where the value of {customer.name} is automatically inserted inside the textbox; the name of the textbox is
 55  * set to match the property name.
 56  * </output>

 58  * @api
 59  */

105     /**
106      * Render the form.
107      *
108      * @param string $action target action
109      * @param array $arguments additional arguments
110      * @param string $controller name of target controller
111      * @param string $package name of target package
112      * @param string $subpackage name of target subpackage
113      * @param mixed $object object to use for the form. Use in conjunction with the "property" attribute on the sub tags
114      * @param string $section The anchor to be added to the action URI (only active if $actionUri is not set)
115      * @param string $format The requested format (e.g. ".html") of the target page (only active if $actionUri is not set)
116      * @param array $additionalParams additional action URI query parameters that won't be prefixed like $arguments (overrule $arguments) (only active if $actionUri is not set)
117      * @param boolean $absolute If set, an absolute action URI is rendered (only active if $actionUri is not set)
118      * @param boolean $addQueryString If set, the current query parameters will be kept in the action URI (only active if $actionUri is not set)
119      * @param array $argumentsToBeExcludedFromQueryString arguments to be removed from the action URI. Only active if $addQueryString = TRUE and $actionUri is not set
120      * @param string $fieldNamePrefix Prefix that will be added to all field names within this form
121      * @param string $actionUri can be used to overwrite the "action" attribute of the form tag
122      * @param string $objectName name of the object that is bound to this form. If this argument is not specified, the name attribute of this form is used to determine the FormObjectName
123      * @param boolean $useParentRequest If set, the parent Request will be used instead ob the current one
124      * @return string rendered form
125      * @api
126      * @throws ViewHelper\Exception
127      */
128     public function render($action = NULL, array $arguments = array(), $controller = NULL, $package = NULL, $subpackage = NULL, $object = NULL, $section = '', $format = '', array $additionalParams = array(), $absolute = FALSE, $addQueryString = FALSE, array $argumentsToBeExcludedFromQueryString = array(), $fieldNamePrefix = NULL, $actionUri = NULL, $objectName = NULL, $useParentRequest = FALSE) {
129         $this->formActionUri = NULL;
130         if ($action === NULL && $actionUri === NULL) {
131             throw new ViewHelper\Exception('FormViewHelper requires "actionUri" or "action" argument to be specified', 1355243748);
132         }
133         $this->tag->addAttribute('action', $this->getFormActionUri());
134 
135         if (strtolower($this->arguments['method']) === 'get') {
136             $this->tag->addAttribute('method', 'get');
137         } else {
138             $this->tag->addAttribute('method', 'post');
139         }
140 
141         $this->addFormObjectNameToViewHelperVariableContainer();
142         $this->addFormObjectToViewHelperVariableContainer();
143         $this->addFieldNamePrefixToViewHelperVariableContainer();
144         $this->addFormFieldNamesToViewHelperVariableContainer();
145         $this->addEmptyHiddenFieldNamesToViewHelperVariableContainer();
146 
147         $formContent = $this->renderChildren();
148 
149         // wrap hidden field in div container in order to create XHTML valid output
150         $content = chr(10) . '<div style="display: none">';
151         if (strtolower($this->arguments['method']) === 'get') {
152             $content .= $this->renderHiddenActionUriQueryParameters();
153         }
154         $content .= $this->renderHiddenIdentityField($this->arguments['object'], $this->getFormObjectName());
155         $content .= $this->renderAdditionalIdentityFields();
156         $content .= $this->renderHiddenReferrerFields();
157         $content .= $this->renderEmptyHiddenFields();
158         // Render the trusted list of all properties after everything else has been rendered
159         $content .= $this->renderTrustedPropertiesField();
160         if (strtolower($this->arguments['method']) !== 'get') {
161             $content .= $this->renderCsrfTokenField();
162         }
163         $content .= chr(10) . '</div>' . chr(10);
164         $content .= $formContent;
165 
166         $this->tag->setContent($content);
167 
168         $this->removeFieldNamePrefixFromViewHelperVariableContainer();
169         $this->removeFormObjectFromViewHelperVariableContainer();
170         $this->removeFormObjectNameFromViewHelperVariableContainer();
171         $this->removeFormFieldNamesFromViewHelperVariableContainer();
172         $this->removeEmptyHiddenFieldNamesFromViewHelperVariableContainer();
173 
174         return $this->tag->render();
175     }

177     /**
178      * Returns the action URI of the form tag.
179      * If the argument "actionUri" is specified, this will be returned
180      * Otherwise this creates the action URI using the UriBuilder
181      *
182      * @return string
183      * @throws ViewHelper\Exception if the action URI could not be created
184      */
185     protected function getFormActionUri() {
186         if ($this->formActionUri !== NULL) {
187             return $this->formActionUri;
188         }
189         if ($this->hasArgument('actionUri')) {
190             $this->formActionUri = $this->arguments['actionUri'];
191         } else {
192             $uriBuilder = $this->controllerContext->getUriBuilder();
193             if ($this->arguments['useParentRequest'] === TRUE) {
194                 $request = $this->controllerContext->getRequest();
195                 if ($request->isMainRequest()) {
196                     throw new ViewHelper\Exception('You can\'t use the parent Request, you are already in the MainRequest.', 1361354942);
197                 }
198                 $uriBuilder = clone $uriBuilder;
199                 $uriBuilder->setRequest($request->getParentRequest());
200             }
201             $uriBuilder
202                 ->reset()
203                 ->setSection($this->arguments['section'])
204                 ->setCreateAbsoluteUri($this->arguments['absolute'])
205                 ->setAddQueryString($this->arguments['addQueryString'])
206                 ->setFormat($this->arguments['format']);
207             if (is_array($this->arguments['additionalParams'])) {
208                 $uriBuilder->setArguments($this->arguments['additionalParams']);
209             }
210             if (is_array($this->arguments['argumentsToBeExcludedFromQueryString'])) {
211                 $uriBuilder->setArgumentsToBeExcludedFromQueryString($this->arguments['argumentsToBeExcludedFromQueryString']);
212             }
213             try {
214                 $this->formActionUri = $uriBuilder
215                     ->uriFor($this->arguments['action'], $this->arguments['arguments'], $this->arguments['controller'], $this->arguments['package'], $this->arguments['subpackage']);
216             } catch (\Exception $exception) {
217                 throw new ViewHelper\Exception($exception->getMessage(), $exception->getCode(), $exception);
218             }
219         }
220         return $this->formActionUri;
221     }

223     /**
224      * Render hidden form fields for query parameters from action URI.
225      * This is only needed if the form method is GET.
226      *
227      * @return string Hidden fields for query parameters from action URI
228      */
229     protected function renderHiddenActionUriQueryParameters() {
230         $result = '';
231         $actionUri = $this->getFormActionUri();
232         $query = parse_url($actionUri, PHP_URL_QUERY);
233 
234         if (is_string($query)) {
235             $queryParts = explode('&', $query);
236             foreach ($queryParts as $queryPart) {
237                 if (strpos($queryPart, '=') !== FALSE) {
238                     list($parameterName, $parameterValue) = explode('=', $queryPart, 2);
239                     $result .= chr(10) . '<input type="hidden" name="' . htmlspecialchars(urldecode($parameterName)) . '" value="' . htmlspecialchars(urldecode($parameterValue)) . '" />';
240                 }
241             }
242         }
243         return $result;
244     }

246     /**
247      * Render additional identity fields which were registered by form elements.
248      * This happens if a form field is defined like property="bla.blubb" - then we might need an identity property for the sub-object "bla".
249      *
250      * @return string HTML-string for the additional identity properties
251      */

253         if ($this->viewHelperVariableContainer->exists('TYPO3\Fluid\ViewHelpers\FormViewHelper', 'additionalIdentityProperties')) {
254             $additionalIdentityProperties = $this->viewHelperVariableContainer->get('TYPO3\Fluid\ViewHelpers\FormViewHelper', 'additionalIdentityProperties');
255             $output = '';
256             foreach ($additionalIdentityProperties as $identity) {
257                 $output .= chr(10) . $identity;
258             }
259             return $output;

264     /**
265      * Renders hidden form fields for referrer information about
266      * the current controller and action.
267      *
268      * @return string Hidden fields with referrer information
269      * @todo filter out referrer information that is equal to the target (e.g. same packageKey)
270      */
271     protected function renderHiddenReferrerFields() {
272         $result = chr(10);
273         $request = $this->controllerContext->getRequest();
274         $argumentNamespace = NULL;
275         if (!$request->isMainRequest()) {
276             $argumentNamespace = $request->getArgumentNamespace();
277 
278             $referrer = array(
279                 '@package' => $request->getControllerPackageKey(),
280                 '@subpackage' => $request->getControllerSubpackageKey(),
281                 '@controller' => $request->getControllerName(),
282                 '@action' => $request->getControllerActionName(),
283                 'arguments' => $this->hashService->appendHmac(base64_encode(serialize($request->getArguments())))
284             );
285             foreach ($referrer as $referrerKey => $referrerValue) {
286                 $referrerValue = \htmlspecialchars($referrerValue);
287                 $result .= '<input type="hidden" name="' . $argumentNamespace . '[__referrer][' . $referrerKey . ']" value="' . $referrerValue . '" />' . chr(10);
288             }
289             $request = $request->getParentRequest();
290         }
291 
292         $arguments = $request->getArguments();
293         if ($argumentNamespace !== NULL && isset($arguments[$argumentNamespace])) {
294             // A sub request was there; thus we can unset the sub requests arguments,
295             // as they are transferred separately via the code block shown above.
296             unset($arguments[$argumentNamespace]);
297         }
298 
299         $referrer = array(
300             '@package' => $request->getControllerPackageKey(),
301             '@subpackage' => $request->getControllerSubpackageKey(),
302             '@controller' => $request->getControllerName(),
303             '@action' => $request->getControllerActionName(),
304             'arguments' => $this->hashService->appendHmac(base64_encode(serialize($arguments)))
305         );
306 
307         foreach ($referrer as $referrerKey => $referrerValue) {
308             $result .= '<input type="hidden" name="__referrer[' . $referrerKey . ']" value="' . htmlspecialchars($referrerValue) . '" />' . chr(10);
309         }
310         return $result;
311     }

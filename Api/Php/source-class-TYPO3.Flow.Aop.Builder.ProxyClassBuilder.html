168     /**
169      * Builds proxy class code which weaves advices into the respective target classes.
170      *
171      * The object configurations provided by the Compiler are searched for possible aspect
172      * annotations. If an aspect class is found, the pointcut expressions are parsed and
173      * a new aspect with one or more advisors is added to the aspect registry of the AOP framework.
174      * Finally all advices are woven into their target classes by generating proxy classes.
175      *
176      * In general, the command typo3.flow:core:compile is responsible for compilation
177      * and calls this method to do so.
178      *
179      * In order to distinguish between an emerged / changed possible target class and
180      * a class which has been matched previously but just didn't have to be proxied,
181      * the latter are kept track of by an "unproxiedClass-*" cache entry.
182      *
183      * @return void
184      */
185     public function build() {
186         $allAvailableClassNamesByPackage = $this->objectManager->getRegisteredClassNames();
187         $possibleTargetClassNames = $this->getProxyableClasses($allAvailableClassNamesByPackage);
188         $actualAspectClassNames = $this->reflectionService->getClassNamesByAnnotation('TYPO3\Flow\Annotations\Aspect');
189         sort($possibleTargetClassNames);
190         sort($actualAspectClassNames);
191 
192         $this->aspectContainers = $this->buildAspectContainers($actualAspectClassNames);
193 
194         $rebuildEverything = FALSE;
195         if ($this->objectConfigurationCache->has('allAspectClassesUpToDate') === FALSE) {
196             $rebuildEverything = TRUE;
197             $this->systemLogger->log('Aspects have been modified, therefore rebuilding all target classes.', LOG_INFO);
198             $this->objectConfigurationCache->set('allAspectClassesUpToDate', TRUE);
199         }
200 
201         $possibleTargetClassNameIndex = new ClassNameIndex();
202         $possibleTargetClassNameIndex->setClassNames($possibleTargetClassNames);
203 
204         $targetClassNameCandidates = new ClassNameIndex();
205         foreach ($this->aspectContainers as $aspectContainer) {
206             $targetClassNameCandidates->applyUnion($aspectContainer->reduceTargetClassNames($possibleTargetClassNameIndex));
207         }
208         $targetClassNameCandidates->sort();
209 
210         foreach ($targetClassNameCandidates->getClassNames() as $targetClassName) {
211             $isUnproxied = $this->objectConfigurationCache->has('unproxiedClass-' . str_replace('\\', '_', $targetClassName));
212             $hasCacheEntry = $this->compiler->hasCacheEntryForClass($targetClassName) || $isUnproxied;
213             if ($rebuildEverything === TRUE || $hasCacheEntry === FALSE) {
214                 $proxyBuildResult = $this->buildProxyClass($targetClassName, $this->aspectContainers);
215                 if ($proxyBuildResult !== FALSE) {
216                     if ($isUnproxied) {
217                         $this->objectConfigurationCache->remove('unproxiedClass-' . str_replace('\\', '_', $targetClassName));
218                     }
219                     $this->systemLogger->log(sprintf('Built AOP proxy for class "%s".', $targetClassName), LOG_DEBUG);
220                 } else {
221                     $this->objectConfigurationCache->set('unproxiedClass-' . str_replace('\\', '_', $targetClassName), TRUE);
222                 }
223             }
224         }
225     }

227     /**
228      * Traverses the aspect containers to find a pointcut from the aspect class name
229      * and pointcut method name
230      *
231      * @param string $aspectClassName Name of the aspect class where the pointcut has been declared
232      * @param string $pointcutMethodName Method name of the pointcut
233      * @return mixed The \TYPO3\Flow\Aop\Pointcut\Pointcut or FALSE if none was found
234      */
235     public function findPointcut($aspectClassName, $pointcutMethodName) {
236         if (!isset($this->aspectContainers[$aspectClassName])) {
237             return FALSE;
238         }
239         foreach ($this->aspectContainers[$aspectClassName]->getPointcuts() as $pointcut) {
240             if ($pointcut->getPointcutMethodName() === $pointcutMethodName) {
241                 return $pointcut;
242             }
243         }
244         return FALSE;

265     /**
266      * Determines which of the given classes are potentially proxyable
267      * and returns their names in an array.
268      *
269      * @param array $classNamesByPackage Names of the classes to check
270      * @return array Names of classes which can be proxied
271      */
272     protected function getProxyableClasses(array $classNamesByPackage) {
273         $proxyableClasses = array();
274         foreach ($classNamesByPackage as $classNames) {
275             foreach ($classNames as $className) {
276                 if (!in_array(substr($className, 0, 15), $this->blacklistedSubPackages)) {
277                     if (!$this->reflectionService->isClassAnnotatedWith($className, 'TYPO3\Flow\Annotations\Aspect') &&
278                         !$this->reflectionService->isClassFinal($className)) {
279                         $proxyableClasses[] = $className;
280                     }
281                 }
282             }
283         }
284         return $proxyableClasses;
285     }

302     /**
303      * Creates and returns an aspect from the annotations found in a class which
304      * is tagged as an aspect. The object acting as an advice will already be
305      * fetched (and therefore instantiated if necessary).
306      *
307      * @param  string $aspectClassName Name of the class which forms the aspect, contains advices etc.
308      * @return mixed The aspect container containing one or more advisors or FALSE if no container could be built
309      * @throws \TYPO3\Flow\Aop\Exception
310      */
311     protected function buildAspectContainer($aspectClassName) {
312         $aspectContainer = new \TYPO3\Flow\Aop\AspectContainer($aspectClassName);
313         $methodNames = get_class_methods($aspectClassName);
314 
315         foreach ($methodNames as $methodName) {
316             foreach ($this->reflectionService->getMethodAnnotations($aspectClassName, $methodName) as $annotation) {
317                 $annotationClass = get_class($annotation);
318                 switch ($annotationClass) {
319                     case 'TYPO3\Flow\Annotations\Around' :
320                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
321                         $advice = new \TYPO3\Flow\Aop\Advice\AroundAdvice($aspectClassName, $methodName);
322                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
323                         $advisor = new \TYPO3\Flow\Aop\Advisor($advice, $pointcut);
324                         $aspectContainer->addAdvisor($advisor);
325                     break;
326                     case 'TYPO3\Flow\Annotations\Before' :
327                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
328                         $advice = new \TYPO3\Flow\Aop\Advice\BeforeAdvice($aspectClassName, $methodName);
329                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
330                         $advisor = new \TYPO3\Flow\Aop\Advisor($advice, $pointcut);
331                         $aspectContainer->addAdvisor($advisor);
332                     break;
333                     case 'TYPO3\Flow\Annotations\AfterReturning' :
334                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
335                         $advice = new \TYPO3\Flow\Aop\Advice\AfterReturningAdvice($aspectClassName, $methodName);
336                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
337                         $advisor = new \TYPO3\Flow\Aop\Advisor($advice, $pointcut);
338                         $aspectContainer->addAdvisor($advisor);
339                     break;
340                     case 'TYPO3\Flow\Annotations\AfterThrowing' :
341                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
342                         $advice = new \TYPO3\Flow\Aop\Advice\AfterThrowingAdvice($aspectClassName, $methodName);
343                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
344                         $advisor = new \TYPO3\Flow\Aop\Advisor($advice, $pointcut);
345                         $aspectContainer->addAdvisor($advisor);
346                     break;
347                     case 'TYPO3\Flow\Annotations\After' :
348                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
349                         $advice = new \TYPO3\Flow\Aop\Advice\AfterAdvice($aspectClassName, $methodName);
350                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
351                         $advisor = new \TYPO3\Flow\Aop\Advisor($advice, $pointcut);
352                         $aspectContainer->addAdvisor($advisor);
353                     break;
354                     case 'TYPO3\Flow\Annotations\Pointcut' :
355                         $pointcutFilterComposite = $this->pointcutExpressionParser->parse($annotation->expression, $this->renderSourceHint($aspectClassName, $methodName, $annotationClass));
356                         $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($annotation->expression, $pointcutFilterComposite, $aspectClassName, $methodName);
357                         $aspectContainer->addPointcut($pointcut);
358                     break;
359                 }
360             }
361         }
362         $introduceAnnotation = $this->reflectionService->getClassAnnotation($aspectClassName, 'TYPO3\Flow\Annotations\Introduce');
363         if ($introduceAnnotation !== NULL) {
364             if ($introduceAnnotation->interfaceName === NULL) {
365                 throw new \TYPO3\Flow\Aop\Exception('The interface introduction in class "' . $aspectClassName . '" does not contain the required interface name).', 1172694761);
366             }
367             $pointcutFilterComposite = $this->pointcutExpressionParser->parse($introduceAnnotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $introduceAnnotation->interfaceName, 'TYPO3\Flow\Annotations\Introduce'));
368             $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($introduceAnnotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
369             $introduction = new \TYPO3\Flow\Aop\InterfaceIntroduction($aspectClassName, $introduceAnnotation->interfaceName, $pointcut);
370             $aspectContainer->addInterfaceIntroduction($introduction);
371         }
372 
373         foreach ($this->reflectionService->getClassPropertyNames($aspectClassName) as $propertyName) {
374             $introduceAnnotation = $this->reflectionService->getPropertyAnnotation($aspectClassName, $propertyName, 'TYPO3\Flow\Annotations\Introduce');
375             if ($introduceAnnotation !== NULL) {
376                 $pointcutFilterComposite = $this->pointcutExpressionParser->parse($introduceAnnotation->pointcutExpression, $this->renderSourceHint($aspectClassName, $propertyName, 'TYPO3\Flow\Annotations\Introduce'));
377                 $pointcut = new \TYPO3\Flow\Aop\Pointcut\Pointcut($introduceAnnotation->pointcutExpression, $pointcutFilterComposite, $aspectClassName);
378                 $introduction = new \TYPO3\Flow\Aop\PropertyIntroduction($aspectClassName, $propertyName, $pointcut);
379                 $aspectContainer->addPropertyIntroduction($introduction);
380             }
381         }
382         if (count($aspectContainer->getAdvisors()) < 1 &&
383             count($aspectContainer->getPointcuts()) < 1 &&
384             count($aspectContainer->getInterfaceIntroductions()) < 1 &&
385             count($aspectContainer->getPropertyIntroductions()) < 1) {
386             throw new \TYPO3\Flow\Aop\Exception('The class "' . $aspectClassName . '" is tagged to be an aspect but doesn\'t contain advices nor pointcut or introduction declarations.', 1169124534);
387         }
388         return $aspectContainer;
389     }

391     /**
392      * Builds methods for a single AOP proxy class for the specified class.
393      *
394      * @param string $targetClassName Name of the class to create a proxy class file for
395      * @param array &$aspectContainers The array of aspect containers from the AOP Framework
396      * @return boolean TRUE if the proxy class could be built, FALSE otherwise.
397      */
398     public function buildProxyClass($targetClassName, array &$aspectContainers) {

580     /**
581      * Traverses all aspect containers, their aspects and their advisors and adds the
582      * methods and their advices to the (usually empty) array of intercepted methods.
583      *
584      * @param array &$interceptedMethods An array (empty or not) which contains the names of the intercepted methods and additional information
585      * @param array $methods An array of class and method names which are matched against the pointcut (class name = name of the class or interface the method was declared)
586      * @param string $targetClassName Name of the class the pointcut should match with
587      * @param array &$aspectContainers All aspects to take into consideration
588      * @return void
589      */
590     protected function addAdvicedMethodsToInterceptedMethods(array &$interceptedMethods, array $methods, $targetClassName, array &$aspectContainers) {
591         $pointcutQueryIdentifier = 0;
592 
593         foreach ($aspectContainers as $aspectContainer) {
594             if (!$aspectContainer->getCachedTargetClassNameCandidates()->hasClassName($targetClassName)) {
595                 continue;
596             }
597             foreach ($aspectContainer->getAdvisors() as $advisor) {
598                 $pointcut = $advisor->getPointcut();
599                 foreach ($methods as $method) {
600                     list($methodDeclaringClassName, $methodName) = $method;
601 
602                     if ($this->reflectionService->isMethodFinal($targetClassName, $methodName)) {
603                         continue;
604                     }
605 
606                     if ($this->reflectionService->isMethodStatic($targetClassName, $methodName)) {
607                         continue;
608                     }
609 
610                     if ($pointcut->matches($targetClassName, $methodName, $methodDeclaringClassName, $pointcutQueryIdentifier)) {
611                         $advice = $advisor->getAdvice();
612                         $interceptedMethods[$methodName]['groupedAdvices'][get_class($advice)][] = array(
613                             'advice' => $advice,
614                             'runtimeEvaluationsClosureCode' => $pointcut->getRuntimeEvaluationsClosureCode()
615                         );
616                         $interceptedMethods[$methodName]['declaringClassName'] = $methodDeclaringClassName;
617                     }
618                     $pointcutQueryIdentifier ++;
619                 }
620             }
621         }
622     }

466             $mockPackage->expects($this->any())->method('getPackageKey')->will($this->returnValue($packageKey));
467             $mockPackage->expects($this->any())->method('getPackageMetaData')->will($this->returnValue($mockMetaData));
468             $unsortedPackages[$packageKey] = $mockPackage;
469         }

472         $packageManager->_set('packages', $unsortedPackages);
473         $packageManager->_set('packageStatesConfiguration', array('packages' => $packages));
474         $packageManager->_call('sortAvailablePackagesByDependencies');
475 
476         /*
477         // There are many "correct" orders of packages. A order is correct if all dependent
478         // packages are ordered before a given package (except for cyclic dependencies).
479         // The following can be used to check that order (but due to cyclic dependencies between
480         // e.g. Flow and Fluid this can not be asserted by default).
481         $newPackageOrder = array_keys($packageManager->_get('packages'));

494         $actualPackageStatesConfiguration = $packageManager->_get('packageStatesConfiguration');
495

 16 /**
 17  * A reader for data placed in "plurals" tag in CLDR.
 18  *
 19  * There are a few similar words used in plurals.xml file of CLDR used by this
 20  * class. Following naming convention is used in the code (a name of
 21  * corresponding tag from xml file is provided in brackets, if any):
 22  * - ruleset: a set of plural rules for a locale [pluralRules]
 23  * - rule: a rule for one of the forms: zero, one, two, few, many [pluralRule]
 24  * - subrule: one of the conditions of rule. One rule can have many conditions
 25  *   joined with "and" or "or" logical operator.
 26  *
 27  * @Flow\Scope("singleton")
 28  * @see http://www.unicode.org/reports/tr35/#Language_Plural_Rules
 29  */
 30 class PluralsReader {

 32     /**
 33      * An expression to catch one plural subrule. One rule consists of one or
 34      * more subrules.
 35      *
 36      * @todo improve the regexp pattern
 37      */
 38     const PATTERN_MATCH_SUBRULE = '/(n|nmod)([0-9]+)?(is|isnot|in|notin|within|notwithin)([0-9]+)(?:\.\.([0-9]+))?(and|or)?/';

 41      * Constants for every plural rule form defined in CLDR.

 43     const RULE_ZERO = 'zero';
 44     const RULE_ONE = 'one';
 45     const RULE_TWO = 'two';
 46     const RULE_FEW = 'few';
 47     const RULE_MANY = 'many';
 48     const RULE_OTHER = 'other';

 60     /**
 61      * An array of rulesets, indexed numerically.
 62      *
 63      * One ruleset contains one or more rules (at most 5, one for every plural
 64      * form - zero, one, two, few, many - a rule 'other' is implicit). There can
 65      * also be NULL ruleset, used by languages which don't have plurals.
 66      *
 67      * A rule is an array with following elements:
 68      * 'modulo' => $x | FALSE,
 69      * 'condition' => array(0 => 'conditionName', 1 => $x, 2 => $y),
 70      * 'logicalOperator' => 'and' | 'or' | FALSE
 71      *
 72      * Legend:
 73      * - if 'modulo' key has an integer value, tested variable (call it $n) has
 74      *   to be replaced with the remainder of division of $n by $x. Otherwise
 75      *   unchanged $n is used for conditional test.
 76      * - 'condition' is an indexed array where first element is a name of test
 77      *   condition (one of: is, isnot, in, notin, within, notwithin). Second
 78      *   element is a value to compare $n with. Third element is optional, and
 79      *   is used only for tests where range is needed (last 4 from the list above)
 80      * - 'logicalOperator' represents a logical operation to be done with next
 81      *   subrule in chain. If current subrule is a last one (or only one), this
 82      *   element is set to FALSE.
 83      *
 84      * @var array
 85      */
 86     protected $rulesets;
 87

 89      * An associative array holding information which ruleset is used by given
 90      * locale. One or more locales can use the same ruleset.
 91      *
 92      * @var array
 93      */
 94     protected $rulesetsIndices;

130     /**
131      * Returns matching plural form based on $quantity and $locale provided.
132      *
133      * Plural form is one of following: zero, one, two, few, many, other.
134      * Last one (other) is returned when number provided doesn't match any
135      * of the rules, or there is no rules for given locale.
136      *
137      * @param mixed $quantity A number to find plural form for (float or int)
138      * @param \TYPO3\Flow\I18n\Locale $locale
139      * @return string One of plural form constants
140      */
141     public function getPluralForm($quantity, \TYPO3\Flow\I18n\Locale $locale) {
142         if (!isset($this->rulesetsIndices[$locale->getLanguage()])) {
143             return self::RULE_OTHER;
144         }
145 
146         $ruleset = $this->rulesets[$locale->getLanguage()][$this->rulesetsIndices[$locale->getLanguage()]];
147 
148         if ($ruleset === NULL) {
149             return self::RULE_OTHER;
150         }
151 
152         foreach ($ruleset as $form => $rule) {
153             foreach ($rule as $subrule) {
154                 $subrulePassed = FALSE;
155 
156                 if ($subrule['modulo'] !== FALSE) {
157                     $quantity = fmod($quantity, $subrule['modulo']);
158                 }
159 
160                 if ($quantity == floor($quantity)) {
161                     $quantity = (int)$quantity;
162                 }
163 
164                 $condition = $subrule['condition'];
165                 switch ($condition[0]) {
166                     case 'is':
167                     case 'isnot':
168                         if (is_int($quantity) && $quantity === $condition[1]) {
169                             $subrulePassed = TRUE;
170                         }
171                         if ($condition[0] === 'isnot') {
172                             $subrulePassed = !$subrulePassed;
173                         }
174                         break;
175                     case 'in':
176                     case 'notin':
177                         if (is_int($quantity) && $quantity >= $condition[1] && $quantity <= $condition[2]) {
178                             $subrulePassed = TRUE;
179                         }
180                         if ($condition[0] === 'notin') {
181                             $subrulePassed = !$subrulePassed;
182                         }
183                         break;
184                     case 'within':
185                     case 'notwithin':
186                         if ($quantity >= $condition[1] && $quantity <= $condition[2]) {
187                             $subrulePassed = TRUE;
188                         }
189                         if ($condition[0] === 'notwithin') {
190                             $subrulePassed = !$subrulePassed;
191                         }
192                         break;
193                 }
194 
195                 if (($subrulePassed && $subrule['logicalOperator'] === 'or') || (!$subrulePassed && $subrule['logicalOperator'] === 'and')) {
196                     break;
197                 }
198             }
199 
200             if ($subrulePassed) {
201                 return $form;
202             }
203         }
204 
205         return self::RULE_OTHER;
206     }

209      * Returns array of plural forms available for particular locale.
210      *
211      * @param \TYPO3\Flow\I18n\Locale $locale Locale to return plural forms for
212      * @return array Plural forms' names (one, zero, two, few, many, other) available for language set in this model

214     public function getPluralForms(\TYPO3\Flow\I18n\Locale $locale) {
215         if (!isset($this->rulesetsIndices[$locale->getLanguage()])) {
216             return array(self::RULE_OTHER);
217         }
218 
219         return array_merge(array_keys($this->rulesets[$locale->getLanguage()][$this->rulesetsIndices[$locale->getLanguage()]]), array(self::RULE_OTHER));
220     }

222     /**
223      * Generates an internal representation of plural rules which can be found
224      * in plurals.xml CLDR file.
225      *
226      * The properties $rulesets and $rulesetsIndices should be empty before
227      * running this method.
228      *
229      * @return void
230      * @see \TYPO3\Flow\I18n\Cldr\Reader\PluralsReader::$rulesets
231      */
232     protected function generateRulesets() {
233         $model = $this->cldrRepository->getModel('supplemental/plurals');
234         $pluralRulesSet = $model->getRawArray('plurals');
235 
236         $index = 0;
237         foreach ($pluralRulesSet as $pluralRulesNodeString => $pluralRules) {
238             $localeLanguages = $model->getAttributeValue($pluralRulesNodeString, 'locales');
239 
240             foreach (explode(' ', $localeLanguages) as $localeLanguage) {
241                 $this->rulesetsIndices[$localeLanguage] = $index;
242             }
243 
244             if (is_array($pluralRules)) {
245                 $ruleset = array();
246                 foreach ($pluralRules as $pluralRuleNodeString => $pluralRule) {
247                     $pluralForm = $model->getAttributeValue($pluralRuleNodeString, 'count');
248                     $ruleset[$pluralForm] = $this->parseRule($pluralRule);
249                 }
250 
251                 foreach (explode(' ', $localeLanguages) as $localeLanguage) {
252                     $this->rulesets[$localeLanguage][$index] = $ruleset;
253                 }
254             }
255 
256             ++$index;
257         }
258     }

260     /**
261      * Parses a plural rule from CLDR.
262      *
263      * A plural rule in CLDR is a string with one or more test conditions, with
264      * 'and' or 'or' logical operators between them. Whole expression can look
265      * like this:
266      *
267      * n is 0 OR n is not 1 AND n mod 100 in 1..19
268      *
269      * As CLDR documentation says, following test conditions can be used:
270      * - is x, is not x: $n is (not) equal $x
271      * - in x..y, not in x..y: $n is (not) one of integers from range <$x, $y>
272      * - within x..y, not within x..y: $n is (not) any number from range <$x, $y>
273      *
274      * Where $n can be a number (also float) as is, or a result of $n mod $x.
275      *
276      * Array returned follows simple internal format (see documentation for
277      * $rulesets property for details).
278      *
279      * @param string $rule
280      * @return array Parsed rule
281      * @throws \TYPO3\Flow\I18n\Cldr\Reader\Exception\InvalidPluralRuleException When plural rule does not match regexp pattern
282      */
283     protected function parseRule($rule) {
284         $parsedRule = array();
285 
286         if (preg_match_all(self::PATTERN_MATCH_SUBRULE, strtolower(str_replace(' ', '', $rule)), $matches, \PREG_SET_ORDER)) {
287             foreach ($matches as $matchedSubrule) {
288                 $subrule = array();
289 
290                 if ($matchedSubrule[1] === 'nmod') {
291                     $subrule['modulo'] = (int)$matchedSubrule[2];
292                 } else {
293                     $subrule['modulo'] = FALSE;
294                 }
295 
296                 $condition = array($matchedSubrule[3], (int)$matchedSubrule[4]);
297                 if (!in_array($matchedSubrule[3], array('is', 'isnot'), TRUE)) {
298                     $condition[2] = (int)$matchedSubrule[5];
299                 }
300 
301                 $subrule['condition'] = $condition;
302 
303                 if (isset($matchedSubrule[6]) && ($matchedSubrule[6] === 'and' || $matchedSubrule[6] === 'or')) {
304                     $subrule['logicalOperator'] = $matchedSubrule[6];
305                 } else {
306                     $subrule['logicalOperator'] = FALSE;
307                 }
308 
309                 $parsedRule[] = $subrule;
310             }
311         } else {
312             throw new \TYPO3\Flow\I18n\Cldr\Reader\Exception\InvalidPluralRuleException('A plural rule string is invalid. CLDR files might be corrupted.', 1275493982);
313         }
314 
315         return $parsedRule;
316     }
